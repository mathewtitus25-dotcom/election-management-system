\chapter{APPENDIX}

\section{Minimum Software Requirements}

\medskip
\textbf{Operating System}: Windows 10/11, macOS, or Linux
\\\\
\textbf{Web Browser}: Google Chrome, Mozilla Firefox, or Microsoft Edge (latest version recommended)
\\\\
\textbf{Backend Framework}: Laravel 12
\\\\
\textbf{Programming Language}: PHP 8.2 or higher
\\\\
\textbf{Database}: MySQL 8.0 or higher
\\\\
\textbf{Web Server}: Apache or Nginx
\\\\
\textbf{Dependency Managers}: Composer (for PHP), NPM (for frontend assets)
\\\\
\textbf{Development Tools (Optional)}: Visual Studio Code, Git

\section{Minimum Hardware Requirements}

\textbf{Processor}: Intel Core i3 / i5 (or AMD equivalent) or higher for smooth system performance.
\\\\
\textbf{RAM}: 4 GB minimum (8 GB recommended for better performance)
\\\\
\textbf{Storage}: At least 1 GB free disk space for project files and database storage.
\\\\
\textbf{Display}: 1366x768 resolution or higher for proper dashboard viewing.
\\\\
\textbf{Input Devices}: Keyboard and mouse for navigation and data entry.
\\\\
\textbf{Network Connectivity}: Stable internet connection for accessing the web-based hospital management system.
\\\\
